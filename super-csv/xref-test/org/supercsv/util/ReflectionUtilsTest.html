177 	/**
178 	 * Tests the findSetter() method by passing primitives and wrapper classes to setters that expect wrapper classes
179 	 * and primitives respectively (this tests the autoboxing logic).
180 	 */
181 	@Test
182 	public void testAutoboxing() throws Exception {
183 		
184 		// first try setting wrapper values onto the primitive setters
185 		
186 		findSetter(bean, "primitiveBoolean", Boolean.class).invoke(bean, Boolean.TRUE);
187 		assertEquals(true, bean.isPrimitiveBoolean());
188 		
189 		findSetter(bean, "primitiveInt", Integer.class).invoke(bean, Integer.valueOf(1));
190 		assertEquals(1, bean.getPrimitiveInt());
191 		
192 		findSetter(bean, "primitiveShort", Short.class).invoke(bean, Short.valueOf("2"));
193 		assertEquals(Short.parseShort("2"), bean.getPrimitiveShort());
194 		
195 		findSetter(bean, "primitiveLong", Long.class).invoke(bean, Long.valueOf("3"));
196 		assertEquals(Long.parseLong("3"), bean.getPrimitiveLong());
197 		
198 		findSetter(bean, "primitiveDouble", Double.class).invoke(bean, Double.valueOf("4.0"));
199 		assertEquals(Double.parseDouble("4.0"), bean.getPrimitiveDouble(), DOUBLE_ASSERT_DELTA);
200 		
201 		findSetter(bean, "primitiveFloat", Float.class).invoke(bean, Float.valueOf("5.0"));
202 		assertEquals(Float.parseFloat("5.0"), bean.getPrimitiveFloat(), DOUBLE_ASSERT_DELTA);
203 		
204 		findSetter(bean, "primitiveChar", Character.class).invoke(bean, Character.valueOf('a'));
205 		assertEquals('a', bean.getPrimitiveChar());
206 		
207 		findSetter(bean, "primitiveByte", Byte.class).invoke(bean, Byte.valueOf("123"));
208 		assertEquals(Byte.parseByte("123"), bean.getPrimitiveByte());
209 		
210 		// now try setting primitive values onto the wrapper setters
211 		
212 		findSetter(bean, "booleanWrapper", boolean.class).invoke(bean, true);
213 		assertEquals(Boolean.TRUE, bean.getBooleanWrapper());
214 		
215 		findSetter(bean, "integerWrapper", int.class).invoke(bean, 1);
216 		assertEquals(Integer.valueOf(1), bean.getIntegerWrapper());
217 		
218 		findSetter(bean, "shortWrapper", short.class).invoke(bean, Short.parseShort("2"));
219 		assertEquals(Short.valueOf("2"), bean.getShortWrapper());
220 		
221 		findSetter(bean, "longWrapper", long.class).invoke(bean, Long.parseLong("3"));
222 		assertEquals(Long.valueOf("3"), bean.getLongWrapper());
223 		
224 		findSetter(bean, "doubleWrapper", double.class).invoke(bean, Double.parseDouble("4.0"));
225 		assertEquals(Double.valueOf("4.0"), bean.getDoubleWrapper());
226 		
227 		findSetter(bean, "floatWrapper", float.class).invoke(bean, Float.parseFloat("5.0"));
228 		assertEquals(Float.valueOf("5.0"), bean.getFloatWrapper());
229 		
230 		findSetter(bean, "charWrapper", char.class).invoke(bean, 'a');
231 		assertEquals(Character.valueOf('a'), bean.getCharWrapper());
232 		
233 		findSetter(bean, "byteWrapper", byte.class).invoke(bean, Byte.parseByte("123"));
234 		assertEquals(Byte.valueOf("123"), bean.getByteWrapper());
235 		
236 	}
