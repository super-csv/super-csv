1   package org.supercsv.prefs;

3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import org.supercsv.cellprocessor.ift.CellProcessor;
7   
8   public class CsvConfig {
9   	
10  	private String[] header;
11  	
12  	private String[] nameMapping;
13  	
14  	private CellProcessor[] processors;
15  	
16  	private CsvConfig(Builder builder){
17  		this.header = builder.headers.toArray(new String[builder.headers.size()]);
18  		this.nameMapping = builder.nameMappings.toArray(new String[builder.nameMappings.size()]);
19  		this.processors = builder.processors.toArray(new CellProcessor[builder.processors.size()]);
20  	}
21  
22  	public String[] getHeader() {
23  		return header;
24  	}
25  
26  	public String[] getNameMapping() {
27  		return nameMapping;
28  	}
29  
30  	public CellProcessor[] getProcessors() {
31  		return processors;
32  	}
33  	
34  	public static class Builder {
35  		
36  		private List<String> headers = new ArrayList<String>();
37  		
38  		private List<String> nameMappings = new ArrayList<String>();
39  		
40  		private List<CellProcessor> processors = new ArrayList<CellProcessor>();
41  		
42  		public Builder column(String header, String nameMapping, CellProcessor processor){
43  			headers.add(header);
44  			nameMappings.add(nameMapping);
45  			processors.add(processor);
46  			return this;
47  		}
48  		
49  		public Builder skipColumn(String header){
50  			headers.add(header);
51  			nameMappings.add(null);
52  			processors.add(null);
53  			return this;
54  		}
55  		
56  		public CsvConfig build(){
57  			return new CsvConfig(this);
58  		}
59  		
60  	}

62  }
