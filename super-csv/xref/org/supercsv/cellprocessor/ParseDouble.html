16  package org.supercsv.cellprocessor;
17  
18  import org.supercsv.cellprocessor.ift.DoubleCellProcessor;
19  import org.supercsv.cellprocessor.ift.StringCellProcessor;
20  import org.supercsv.exception.SuperCsvCellProcessorException;
21  import org.supercsv.util.CsvContext;
22  
23  /**
24   * Converts a String to a Double.
25   * 
26   * @author Kasper B. Graversen
27   */
28  public class ParseDouble extends CellProcessorAdaptor implements StringCellProcessor {
29  	
30  	/**
31  	 * Constructs a new <tt>ParseDouble</tt> processor, which converts a String to a Double.
32  	 */
33  	public ParseDouble() {
34  		super();
35  	}
36  	
37  	/**
38  	 * Constructs a new <tt>ParseDouble</tt> processor, which converts a String to a Double, then calls the next
39  	 * processor in the chain.
40  	 * 
41  	 * @param next
42  	 *            the next processor in the chain
43  	 * @throws NullPointerException
44  	 *             if next is null
45  	 */
46  	public ParseDouble(final DoubleCellProcessor next) {
47  		super(next);
48  	}
49  	
50  	/**
51  	 * {@inheritDoc}
52  	 * 
53  	 * @throws SuperCsvCellProcessorException
54  	 *             if value is null, isn't a Double or String, or can't be parsed as a Double
55  	 */
56  	public Object execute(final Object value, final CsvContext context) {
57  		validateInputNotNull(value, context);
58  		
59  		final Double result;
60  		if( value instanceof Double ) {
61  			result = (Double) value;
62  		} else if( value instanceof String ) {
63  			try {
64  				result = new Double((String) value);
65  			}
66  			catch(final NumberFormatException e) {
67  				throw new SuperCsvCellProcessorException(String.format("'%s' could not be parsed as a Double", value),
68  					context, this, e);
69  			}
70  		} else {
71  			final String actualClassName = value.getClass().getName();
72  			throw new SuperCsvCellProcessorException(String.format(
73  				"the input value should be of type Double or String but is of type %s", actualClassName), context, this);
74  		}
75  		
76  		return next.execute(result, context);
77  	}
78  }
